21   <pre><span class="s0">&quot;&quot;&quot;distutils.cygwinccompiler 
22    
23   Provides the CygwinCCompiler class, a subclass of UnixCCompiler that 
24   handles the Cygwin port of the GNU C compiler to Windows.  It also contains 
25   the Mingw32CCompiler class which handles the mingw32 port of GCC (same as 
26   cygwin in no-cygwin mode). 
27   &quot;&quot;&quot;</span>
28   
29   <span class="s2"># problems:</span>
30   <span class="s2">#</span>
31   <span class="s2"># * if you use a msvc compiled python version (1.5.2)</span>
32   <span class="s2">#   1. you have to insert a __GNUC__ section in its config.h</span>
33   <span class="s2">#   2. you have to generate an import library for its dll</span>
34   <span class="s2">#      - create a def-file for python??.dll</span>
35   <span class="s2">#      - create an import library using</span>
36   <span class="s2">#             dlltool --dllname python15.dll --def python15.def \</span>
37   <span class="s2">#                       --output-lib libpython15.a</span>
38   <span class="s2">#</span>
39   <span class="s2">#   see also http://starship.python.net/crew/kernr/mingw32/Notes.html</span>
40   <span class="s2">#</span>
41   <span class="s2"># * We put export_symbols in a def-file, and don't use</span>
42   <span class="s2">#   --export-all-symbols because it doesn't worked reliable in some</span>
43   <span class="s2">#   tested configurations. And because other windows compilers also</span>
44   <span class="s2">#   need their symbols specified this no serious problem.</span>
45   <span class="s2">#</span>
46   <span class="s2"># tested configurations:</span>
47   <span class="s2">#</span>
48   <span class="s2"># * cygwin gcc 2.91.57/ld 2.9.4/dllwrap 0.2.4 works</span>
49   <span class="s2">#   (after patching python's config.h and for C++ some other include files)</span>
50   <span class="s2">#   see also http://starship.python.net/crew/kernr/mingw32/Notes.html</span>
51   <span class="s2"># * mingw32 gcc 2.95.2/ld 2.9.4/dllwrap 0.2.4 works</span>
52   <span class="s2">#   (ld doesn't support -shared, so we use dllwrap)</span>
53   <span class="s2"># * cygwin gcc 2.95.2/ld 2.10.90/dllwrap 2.10.90 works now</span>
54   <span class="s2">#   - its dllwrap doesn't work, there is a bug in binutils 2.10.90</span>
55   <span class="s2">#     see also http://sources.redhat.com/ml/cygwin/2000-06/msg01274.html</span>
56   <span class="s2">#   - using gcc -mdll instead dllwrap doesn't work without -static because</span>
57   <span class="s2">#     it tries to link against dlls instead their import libraries. (If</span>
58   <span class="s2">#     it finds the dll first.)</span>
59   <span class="s2">#     By specifying -static we force ld to link against the import libraries,</span>
60   <span class="s2">#     this is windows standard and there are normally not the necessary symbols</span>
61   <span class="s2">#     in the dlls.</span>
62   <span class="s2">#   *** only the version of June 2000 shows these problems</span>
63   <span class="s2"># * cygwin gcc 3.2/ld 2.13.90 works</span>
64   <span class="s2">#   (ld supports -shared)</span>
65   <span class="s2"># * mingw gcc 3.2/ld 2.13 works</span>
66   <span class="s2">#   (ld supports -shared)</span>
67   <span class="s2"># * llvm-mingw with Clang 11 works</span>
68   <span class="s2">#   (lld supports -shared)</span>

325  <span class="s3">def </span><span class="s1">check_config_h():</span>
326      <span class="s0">&quot;&quot;&quot;Check if the current Python installation appears amenable to building 
327      extensions with GCC. 
328   
329      Returns a tuple (status, details), where 'status' is one of the following 
330      constants: 
331   
332      - CONFIG_H_OK: all is well, go ahead and compile 
333      - CONFIG_H_NOTOK: doesn't look good 
334      - CONFIG_H_UNCERTAIN: not sure -- unable to read pyconfig.h 
335   
336      'details' is a human-readable string explaining the situation. 
337   
338      Note there are two ways to conclude &quot;OK&quot;: either 'sys.version' contains 
339      the string &quot;GCC&quot; (implying that this Python was built with GCC), or the 
340      installed &quot;pyconfig.h&quot; contains the string &quot;__GNUC__&quot;. 
341      &quot;&quot;&quot;</span>
342  
343      <span class="s2"># XXX since this function also checks sys.version, it's not strictly a</span>
344      <span class="s2"># &quot;pyconfig.h&quot; check -- should probably be renamed...</span>
345  
346      <span class="s3">from </span><span class="s1">distutils </span><span class="s3">import </span><span class="s1">sysconfig</span>
347  
348      <span class="s2"># if sys.version contains GCC then python was compiled with GCC, and the</span>
349      <span class="s2"># pyconfig.h file should be OK</span>
350      <span class="s3">if </span><span class="s4">&quot;GCC&quot; </span><span class="s3">in </span><span class="s1">sys.version:</span>
351          <span class="s3">return </span><span class="s1">CONFIG_H_OK</span><span class="s3">, </span><span class="s4">&quot;sys.version mentions 'GCC'&quot;</span>
352  
353      <span class="s2"># Clang would also work</span>
354      <span class="s3">if </span><span class="s4">&quot;Clang&quot; </span><span class="s3">in </span><span class="s1">sys.version:</span>
355          <span class="s3">return </span><span class="s1">CONFIG_H_OK</span><span class="s3">, </span><span class="s4">&quot;sys.version mentions 'Clang'&quot;</span>
356  
357      <span class="s2"># let's see if __GNUC__ is mentioned in python.h</span>
358      <span class="s1">fn = sysconfig.get_config_h_filename()</span>
359      <span class="s3">try</span><span class="s1">:</span>
360          <span class="s1">config_h = open(fn)</span>
361          <span class="s3">try</span><span class="s1">:</span>
362              <span class="s3">if </span><span class="s4">&quot;__GNUC__&quot; </span><span class="s3">in </span><span class="s1">config_h.read():</span>
363                  <span class="s3">return </span><span class="s1">CONFIG_H_OK</span><span class="s3">, </span><span class="s4">&quot;'%s' mentions '__GNUC__'&quot; </span><span class="s1">% fn</span>
364              <span class="s3">else</span><span class="s1">:</span>
365                  <span class="s3">return </span><span class="s1">CONFIG_H_NOTOK</span><span class="s3">, </span><span class="s4">&quot;'%s' does not mention '__GNUC__'&quot; </span><span class="s1">% fn</span>
366          <span class="s3">finally</span><span class="s1">:</span>
367              <span class="s1">config_h.close()</span>
368      <span class="s3">except </span><span class="s1">OSError </span><span class="s3">as </span><span class="s1">exc:</span>
369          <span class="s3">return </span><span class="s1">(CONFIG_H_UNCERTAIN</span><span class="s3">,</span>
370                  <span class="s4">&quot;couldn't read '%s': %s&quot; </span><span class="s1">% (fn</span><span class="s3">, </span><span class="s1">exc.strerror))</span>
