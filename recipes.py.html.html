338              <span class="s1">pending -= </span><span class="s4">1</span>

343      <span class="s0">&quot;&quot;&quot; 
344      Returns a 2-tuple of iterables derived from the input iterable. 
345      The first yields the items that have ``pred(item) == False``. 
346      The second yields the items that have ``pred(item) == True``. 
347   
348          &gt;&gt;&gt; is_odd = lambda x: x % 2 != 0 
349          &gt;&gt;&gt; iterable = range(10)

351          &gt;&gt;&gt; list(even_items), list(odd_items)

355   
356          &gt;&gt;&gt; iterable = [0, 1, False, True, '', ' '] 
357          &gt;&gt;&gt; false_items, true_items = partition(None, iterable) 
358          &gt;&gt;&gt; list(false_items), list(true_items)

363          <span class="s1">pred = bool</span>

366      <span class="s1">t1</span><span class="s2">, </span><span class="s1">t2 = tee(evaluations)</span>
367      <span class="s2">return </span><span class="s1">(</span>
368          <span class="s1">(x </span><span class="s2">for </span><span class="s1">(cond</span><span class="s2">, </span><span class="s1">x) </span><span class="s2">in </span><span class="s1">t1 </span><span class="s2">if not </span><span class="s1">cond)</span><span class="s2">,</span>

373  <span class="s2">def </span><span class="s1">powerset(iterable):</span>
374      <span class="s0">&quot;&quot;&quot;Yields all possible subsets of the iterable. 
375   
376          &gt;&gt;&gt; list(powerset([1, 2, 3])) 
377          [(), (1,), (2,), (3,), (1, 2), (1, 3), (2, 3), (1, 2, 3)] 
378   
379      :func:`powerset` will operate on iterables that aren't :class:`set` 
380      instances, so repeated elements in the input will produce repeated elements 
381      in the output. Use :func:`unique_everseen` on the input to avoid generating 
382      duplicates: 
383

398      Yield unique elements, preserving order. 
399   
400          &gt;&gt;&gt; list(unique_everseen('AAAABBBCCDAABBB'))

403          ['A', 'B', 'C', 'D'] 
404   
405      Sequences with a mix of hashable and unhashable items can be used. 
406      The function will be slower (i.e., `O(n^2)`) for unhashable items. 
407   
408      Remember that ``list`` objects are unhashable - you can use the *key* 
409      parameter to transform the list to a tuple (which is hashable) to 
410      avoid a slowdown. 
411   
412          &gt;&gt;&gt; iterable = ([1, 2], [2, 3], [1, 2]) 
413          &gt;&gt;&gt; list(unique_everseen(iterable))  # Slow 
414          [[1, 2], [2, 3]] 
415          &gt;&gt;&gt; list(unique_everseen(iterable, key=tuple))  # Faster 
416          [[1, 2], [2, 3]] 
417   
418      Similary, you may want to convert unhashable ``set`` objects with 
419      ``key=frozenset``. For ``dict`` objects, 
420      ``key=lambda x: frozenset(x.items())`` can be used. 
421   
422      &quot;&quot;&quot;</span>
423      <span class="s1">seenset = set()</span>
424      <span class="s1">seenset_add = seenset.add</span>
425      <span class="s1">seenlist = []</span>
426      <span class="s1">seenlist_add = seenlist.append</span>
427      <span class="s1">use_key = key </span><span class="s2">is not None</span>
428  
429      <span class="s2">for </span><span class="s1">element </span><span class="s2">in </span><span class="s1">iterable:</span>
430          <span class="s1">k = key(element) </span><span class="s2">if </span><span class="s1">use_key </span><span class="s2">else </span><span class="s1">element</span>
431          <span class="s2">try</span><span class="s1">:</span>
432              <span class="s2">if </span><span class="s1">k </span><span class="s2">not in </span><span class="s1">seenset:</span>
433                  <span class="s1">seenset_add(k)</span>
434                  <span class="s2">yield </span><span class="s1">element</span>
435          <span class="s2">except </span><span class="s1">TypeError:</span>
436              <span class="s2">if </span><span class="s1">k </span><span class="s2">not in </span><span class="s1">seenlist:</span>
437                  <span class="s1">seenlist_add(k)</span>
438                  <span class="s2">yield </span><span class="s1">element</span>

565  <span class="s2">def </span><span class="s1">nth_combination(iterable</span><span class="s2">, </span><span class="s1">r</span><span class="s2">, </span><span class="s1">index):</span>
566      <span class="s0">&quot;&quot;&quot;Equivalent to ``list(combinations(iterable, r))[index]``. 
567   
568      The subsequences of *iterable* that are of length *r* can be ordered 
569      lexicographically. :func:`nth_combination` computes the subsequence at 
570      sort position *index* directly, without computing the previous 
571      subsequences.

574          (0, 3, 4)

577      of *iterable*. 
578      ``IndexError`` will be raised if the given *index* is invalid. 
579      &quot;&quot;&quot;</span>
580      <span class="s1">pool = tuple(iterable)</span>
581      <span class="s1">n = len(pool)</span>
582      <span class="s2">if </span><span class="s1">(r &lt; </span><span class="s4">0</span><span class="s1">) </span><span class="s2">or </span><span class="s1">(r &gt; n):</span>
583          <span class="s2">raise </span><span class="s1">ValueError</span>
584  
585      <span class="s1">c = </span><span class="s4">1</span>
586      <span class="s1">k = min(r</span><span class="s2">, </span><span class="s1">n - r)</span>
587      <span class="s2">for </span><span class="s1">i </span><span class="s2">in </span><span class="s1">range(</span><span class="s4">1</span><span class="s2">, </span><span class="s1">k + </span><span class="s4">1</span><span class="s1">):</span>
588          <span class="s1">c = c * (n - k + i) // i</span>
589  
590      <span class="s2">if </span><span class="s1">index &lt; </span><span class="s4">0</span><span class="s1">:</span>
591          <span class="s1">index += c</span>
592  
593      <span class="s2">if </span><span class="s1">(index &lt; </span><span class="s4">0</span><span class="s1">) </span><span class="s2">or </span><span class="s1">(index &gt;= c):</span>
594          <span class="s2">raise </span><span class="s1">IndexError</span>
595  
596      <span class="s1">result = []</span>
597      <span class="s2">while </span><span class="s1">r:</span>
598          <span class="s1">c</span><span class="s2">, </span><span class="s1">n</span><span class="s2">, </span><span class="s1">r = c * r // n</span><span class="s2">, </span><span class="s1">n - </span><span class="s4">1</span><span class="s2">, </span><span class="s1">r - </span><span class="s4">1</span>

600              <span class="s1">index -= c</span>
601              <span class="s1">c</span><span class="s2">, </span><span class="s1">n = c * (n - r) // n</span><span class="s2">, </span><span class="s1">n - </span><span class="s4">1</span>
602          <span class="s1">result.append(pool[-</span><span class="s4">1 </span><span class="s1">- n])</span>
603  
604      <span class="s2">return </span><span class="s1">tuple(result)</span>
