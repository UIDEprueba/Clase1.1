38    
39       If the directory already exists (or if 'name' is the empty string, which 
40       means the current directory, which of course exists), then do nothing. 
41       Raise DistutilsFileError if unable to create some directory along the way 
42       (eg. some sub-path exists, but is a file rather than a directory). 
43       If 'verbose' is true, print a one-line summary of each mkdir to stdout. 
44       Return the list of directories actually created. 
45       &quot;&quot;&quot;</span>
46

52                 <span class="s5">&quot;mkpath: 'name' must be a string (got %r)&quot; </span><span class="s1">% (name</span><span class="s2">,</span><span class="s1">))</span>
53   
54       <span class="s3"># XXX what's the better way to handle verbosity? print as we create</span>
55       <span class="s3"># each directory in the path (the current behaviour), or only announce</span>
56       <span class="s3"># the creation of the whole path? (quite easy to do the latter since</span>
57       <span class="s3"># we're not using a recursive algorithm)</span>
58   
59       <span class="s1">name = os.path.normpath(name)</span>
60       <span class="s1">created_dirs = []</span>
61       <span class="s2">if </span><span class="s1">os.path.isdir(name) </span><span class="s2">or </span><span class="s1">name == </span><span class="s5">''</span><span class="s1">:</span>
62           <span class="s2">return </span><span class="s1">created_dirs</span>
63       <span class="s2">if </span><span class="s1">_path_created.get(os.path.abspath(name)):</span>
64           <span class="s2">return </span><span class="s1">created_dirs</span>
65   
66       <span class="s1">(head</span><span class="s2">, </span><span class="s1">tail) = os.path.split(name)</span>
67       <span class="s1">tails = [tail]                      </span><span class="s3"># stack of lone dirs to create</span>
68   
69       <span class="s2">while </span><span class="s1">head </span><span class="s2">and </span><span class="s1">tail </span><span class="s2">and not </span><span class="s1">os.path.isdir(head):</span>
70           <span class="s1">(head</span><span class="s2">, </span><span class="s1">tail) = os.path.split(head)</span>
71           <span class="s1">tails.insert(</span><span class="s4">0</span><span class="s2">, </span><span class="s1">tail)          </span><span class="s3"># push next higher dir onto stack</span>
72   
73       <span class="s3"># now 'head' contains the deepest directory that already exists</span>
74       <span class="s3"># (that is, the child of 'head' in 'name' is the highest directory</span>
75       <span class="s3"># that does *not* exist)</span>
76       <span class="s2">for </span><span class="s1">d </span><span class="s2">in </span><span class="s1">tails:</span>
77           <span class="s3">#print &quot;head = %s, d = %s: &quot; % (head, d),</span>
78           <span class="s1">head = os.path.join(head</span><span class="s2">, </span><span class="s1">d)</span>
79           <span class="s1">abs_head = os.path.abspath(head)</span>
80   
81           <span class="s2">if </span><span class="s1">_path_created.get(abs_head):</span>
82               <span class="s2">continue</span>
83   
84           <span class="s2">if </span><span class="s1">verbose &gt;= </span><span class="s4">1</span><span class="s1">:</span>
85               <span class="s1">log.info(</span><span class="s5">&quot;creating %s&quot;</span><span class="s2">, </span><span class="s1">head)</span>
86   
87           <span class="s2">if not </span><span class="s1">dry_run:</span>
88               <span class="s2">try</span><span class="s1">:</span>
89                   <span class="s1">os.mkdir(head</span><span class="s2">, </span><span class="s1">mode)</span>
90               <span class="s2">except </span><span class="s1">OSError </span><span class="s2">as </span><span class="s1">exc:</span>
91                   <span class="s2">if not </span><span class="s1">(exc.errno == errno.EEXIST </span><span class="s2">and </span><span class="s1">os.path.isdir(head)):</span>
92                       <span class="s2">raise </span><span class="s1">DistutilsFileError(</span>
93                             <span class="s5">&quot;could not create '%s': %s&quot; </span><span class="s1">% (head</span><span class="s2">, </span><span class="s1">exc.args[-</span><span class="s4">1</span><span class="s1">]))</span>
94               <span class="s1">created_dirs.append(head)</span>
95   
96           <span class="s1">_path_created[abs_head] = </span><span class="s4">1</span>
97       <span class="s2">return </span><span class="s1">created_dirs</span>

99   <span class="s2">def </span><span class="s1">create_tree(base_dir</span><span class="s2">, </span><span class="s1">files</span><span class="s2">, </span><span class="s1">mode=</span><span class="s4">0o777</span><span class="s2">, </span><span class="s1">verbose=</span><span class="s4">1</span><span class="s2">, </span><span class="s1">dry_run=</span><span class="s4">0</span><span class="s1">):</span>
100      <span class="s0">&quot;&quot;&quot;Create all the empty directories under 'base_dir' needed to put 'files' 
101      there. 
102   
103      'base_dir' is just the name of a directory which doesn't necessarily 
104      exist yet; 'files' is a list of filenames to be interpreted relative to 
105      'base_dir'.  'base_dir' + the directory portion of every file in 'files'

113  
114      <span class="s3"># Now create them</span>
115      <span class="s2">for </span><span class="s1">dir </span><span class="s2">in </span><span class="s1">sorted(need_dir):</span>
116          <span class="s1">mkpath(dir</span><span class="s2">, </span><span class="s1">mode</span><span class="s2">, </span><span class="s1">verbose=verbose</span><span class="s2">, </span><span class="s1">dry_run=dry_run)</span>

118  <span class="s2">def </span><span class="s1">copy_tree(src</span><span class="s2">, </span><span class="s1">dst</span><span class="s2">, </span><span class="s1">preserve_mode=</span><span class="s4">1</span><span class="s2">, </span><span class="s1">preserve_times=</span><span class="s4">1</span><span class="s2">,</span>
119                <span class="s1">preserve_symlinks=</span><span class="s4">0</span><span class="s2">, </span><span class="s1">update=</span><span class="s4">0</span><span class="s2">, </span><span class="s1">verbose=</span><span class="s4">1</span><span class="s2">, </span><span class="s1">dry_run=</span><span class="s4">0</span><span class="s1">):</span>
120      <span class="s0">&quot;&quot;&quot;Copy an entire directory tree 'src' to a new location 'dst'. 
121   
122      Both 'src' and 'dst' must be directory names.  If 'src' is not a 
123      directory, raise DistutilsFileError.  If 'dst' does not exist, it is 
124      created with 'mkpath()'.  The end result of the copy is that every 
125      file in 'src' is copied to 'dst', and directories under 'src' are 
126      recursively copied to 'dst'.  Return the list of files that were 
127      copied or might have been copied, using their output name.  The 
128      return value is unaffected by 'update' or 'dry_run': it is simply 
129      the list of all files under 'src', with the names changed to be 
130      under 'dst'. 
131   
132      'preserve_mode' and 'preserve_times' are the same as for 
133      'copy_file'; note that they only apply to regular files, not to 
134      directories.  If 'preserve_symlinks' is true, symlinks will be 
135      copied as symlinks (on platforms that support them!); otherwise 
136      (the default), the destination of the symlink will be copied. 
137      'update' and 'verbose' are the same as for 'copy_file'. 
138      &quot;&quot;&quot;</span>
139      <span class="s2">from </span><span class="s1">distutils.file_util </span><span class="s2">import </span><span class="s1">copy_file</span>
140  
141      <span class="s2">if not </span><span class="s1">dry_run </span><span class="s2">and not </span><span class="s1">os.path.isdir(src):</span>
142          <span class="s2">raise </span><span class="s1">DistutilsFileError(</span>
143                <span class="s5">&quot;cannot copy tree '%s': not a directory&quot; </span><span class="s1">% src)</span>
144      <span class="s2">try</span><span class="s1">:</span>
145          <span class="s1">names = os.listdir(src)</span>
146      <span class="s2">except </span><span class="s1">OSError </span><span class="s2">as </span><span class="s1">e:</span>
147          <span class="s2">if </span><span class="s1">dry_run:</span>
148              <span class="s1">names = []</span>
149          <span class="s2">else</span><span class="s1">:</span>
150              <span class="s2">raise </span><span class="s1">DistutilsFileError(</span>
151                    <span class="s5">&quot;error listing files in '%s': %s&quot; </span><span class="s1">% (src</span><span class="s2">, </span><span class="s1">e.strerror))</span>
152  
153      <span class="s2">if not </span><span class="s1">dry_run:</span>
154          <span class="s1">mkpath(dst</span><span class="s2">, </span><span class="s1">verbose=verbose)</span>
155  
156      <span class="s1">outputs = []</span>
157  
158      <span class="s2">for </span><span class="s1">n </span><span class="s2">in </span><span class="s1">names:</span>
159          <span class="s1">src_name = os.path.join(src</span><span class="s2">, </span><span class="s1">n)</span>
160          <span class="s1">dst_name = os.path.join(dst</span><span class="s2">, </span><span class="s1">n)</span>
161  
162          <span class="s2">if </span><span class="s1">n.startswith(</span><span class="s5">'.nfs'</span><span class="s1">):</span>
163              <span class="s3"># skip NFS rename files</span>
164              <span class="s2">continue</span>
165  
166          <span class="s2">if </span><span class="s1">preserve_symlinks </span><span class="s2">and </span><span class="s1">os.path.islink(src_name):</span>
167              <span class="s1">link_dest = os.readlink(src_name)</span>
168              <span class="s2">if </span><span class="s1">verbose &gt;= </span><span class="s4">1</span><span class="s1">:</span>
169                  <span class="s1">log.info(</span><span class="s5">&quot;linking %s -&gt; %s&quot;</span><span class="s2">, </span><span class="s1">dst_name</span><span class="s2">, </span><span class="s1">link_dest)</span>
170              <span class="s2">if not </span><span class="s1">dry_run:</span>
171                  <span class="s1">os.symlink(link_dest</span><span class="s2">, </span><span class="s1">dst_name)</span>
172              <span class="s1">outputs.append(dst_name)</span>
173  
174          <span class="s2">elif </span><span class="s1">os.path.isdir(src_name):</span>
175              <span class="s1">outputs.extend(</span>
176                  <span class="s1">copy_tree(src_name</span><span class="s2">, </span><span class="s1">dst_name</span><span class="s2">, </span><span class="s1">preserve_mode</span><span class="s2">,</span>
177                            <span class="s1">preserve_times</span><span class="s2">, </span><span class="s1">preserve_symlinks</span><span class="s2">, </span><span class="s1">update</span><span class="s2">,</span>
178                            <span class="s1">verbose=verbose</span><span class="s2">, </span><span class="s1">dry_run=dry_run))</span>
179          <span class="s2">else</span><span class="s1">:</span>
180              <span class="s1">copy_file(src_name</span><span class="s2">, </span><span class="s1">dst_name</span><span class="s2">, </span><span class="s1">preserve_mode</span><span class="s2">,</span>
181                        <span class="s1">preserve_times</span><span class="s2">, </span><span class="s1">update</span><span class="s2">, </span><span class="s1">verbose=verbose</span><span class="s2">,</span>
182                        <span class="s1">dry_run=dry_run)</span>
183              <span class="s1">outputs.append(dst_name)</span>
184  
185      <span class="s2">return </span><span class="s1">outputs</span>
