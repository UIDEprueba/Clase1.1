25   but are necessary to get them to work. They include a whole bunch of low-level

232  <span class="s2">def </span><span class="s1">_temporary_keychain():</span>
233      <span class="s0">&quot;&quot;&quot; 
234      This function creates a temporary Mac keychain that we can use to work with 
235      credentials. This keychain uses a one-time password and a temporary file to 
236      store the data. We expect to have one keychain per socket. The returned 
237      SecKeychainRef must be freed by the caller, including calling 
238      SecKeychainDelete. 
239   
240      Returns a tuple of the SecKeychainRef and the path to the temporary 
241      directory that contains it. 
242      &quot;&quot;&quot;</span>
243      <span class="s3"># Unfortunately, SecKeychainCreate requires a path to a keychain. This</span>
244      <span class="s3"># means we cannot use mkstemp to use a generic temporary file. Instead,</span>
245      <span class="s3"># we're going to create a temporary directory and a filename to use there.</span>
246      <span class="s3"># This filename will be 8 random bytes expanded into base64. We also need</span>
247      <span class="s3"># some random bytes to password-protect the keychain we're creating, so we</span>
248      <span class="s3"># ask for 40 random bytes.</span>
249      <span class="s1">random_bytes = os.urandom(</span><span class="s5">40</span><span class="s1">)</span>
250      <span class="s1">filename = base64.b16encode(random_bytes[:</span><span class="s5">8</span><span class="s1">]).decode(</span><span class="s6">&quot;utf-8&quot;</span><span class="s1">)</span>

252      <span class="s1">tempdirectory = tempfile.mkdtemp()</span>
253  
254      <span class="s1">keychain_path = os.path.join(tempdirectory</span><span class="s2">, </span><span class="s1">filename).encode(</span><span class="s6">&quot;utf-8&quot;</span><span class="s1">)</span>
255  
256      <span class="s3"># We now want to create the keychain itself.</span>
257      <span class="s1">keychain = Security.SecKeychainRef()</span>
258      <span class="s1">status = Security.SecKeychainCreate(</span>
259          <span class="s1">keychain_path</span><span class="s2">, </span><span class="s1">len(password)</span><span class="s2">, </span><span class="s1">password</span><span class="s2">, False, None, </span><span class="s1">ctypes.byref(keychain)</span>

262  
263      <span class="s3"># Having created the keychain, we want to pass it off to the caller.</span>
264      <span class="s2">return </span><span class="s1">keychain</span><span class="s2">, </span><span class="s1">tempdirectory</span>

322  <span class="s2">def </span><span class="s1">_load_client_cert_chain(keychain</span><span class="s2">, </span><span class="s1">*paths):</span>
323      <span class="s0">&quot;&quot;&quot; 
324      Load certificates and maybe keys from a number of files. Has the end goal 
325      of returning a CFArray containing one SecIdentityRef, and then zero or more 
326      SecCertificateRef objects, suitable for use as a client certificate trust 
327      chain. 
328      &quot;&quot;&quot;</span>
329      <span class="s3"># Ok, the strategy.</span>
330      <span class="s3">#</span>
331      <span class="s3"># This relies on knowing that macOS will not give you a SecIdentityRef</span>
332      <span class="s3"># unless you have imported a key into a keychain. This is a somewhat</span>
333      <span class="s3"># artificial limitation of macOS (for example, it doesn't necessarily</span>
334      <span class="s3"># affect iOS), but there is nothing inside Security.framework that lets you</span>
335      <span class="s3"># get a SecIdentityRef without having a key in a keychain.</span>
336      <span class="s3">#</span>
337      <span class="s3"># So the policy here is we take all the files and iterate them in order.</span>
338      <span class="s3"># Each one will use SecItemImport to have one or more objects loaded from</span>
339      <span class="s3"># it. We will also point at a keychain that macOS can use to work with the</span>
340      <span class="s3"># private key.</span>
341      <span class="s3">#</span>
342      <span class="s3"># Once we have all the objects, we'll check what we actually have. If we</span>
343      <span class="s3"># already have a SecIdentityRef in hand, fab: we'll use that. Otherwise,</span>
344      <span class="s3"># we'll take the first certificate (which we assume to be our leaf) and</span>
345      <span class="s3"># ask the keychain to give us a SecIdentityRef with that cert's associated</span>
346      <span class="s3"># key.</span>
347      <span class="s3">#</span>
348      <span class="s3"># We'll then return a CFArray containing the trust chain: one</span>
349      <span class="s3"># SecIdentityRef and then zero-or-more SecCertificateRef objects. The</span>
350      <span class="s3"># responsibility for freeing this CFArray will be with the caller. This</span>
351      <span class="s3"># CFArray must remain alive for the entire connection, so in practice it</span>
352      <span class="s3"># will be stored with a single SSLSocket, along with the reference to the</span>
353      <span class="s3"># keychain.</span>
354      <span class="s1">certificates = []</span>
355      <span class="s1">identities = []</span>
356  
357      <span class="s3"># Filter out bad paths.</span>
358      <span class="s1">paths = (path </span><span class="s2">for </span><span class="s1">path </span><span class="s2">in </span><span class="s1">paths </span><span class="s2">if </span><span class="s1">path)</span>
359  
360      <span class="s2">try</span><span class="s1">:</span>
361          <span class="s2">for </span><span class="s1">file_path </span><span class="s2">in </span><span class="s1">paths:</span>
362              <span class="s1">new_identities</span><span class="s2">, </span><span class="s1">new_certs = _load_items_from_file(keychain</span><span class="s2">, </span><span class="s1">file_path)</span>
363              <span class="s1">identities.extend(new_identities)</span>
364              <span class="s1">certificates.extend(new_certs)</span>
365  
366          <span class="s3"># Ok, we have everything. The question is: do we have an identity? If</span>
367          <span class="s3"># not, we want to grab one from the first cert we have.</span>
368          <span class="s2">if not </span><span class="s1">identities:</span>
369              <span class="s1">new_identity = Security.SecIdentityRef()</span>
370              <span class="s1">status = Security.SecIdentityCreateWithCertificate(</span>
371                  <span class="s1">keychain</span><span class="s2">, </span><span class="s1">certificates[</span><span class="s5">0</span><span class="s1">]</span><span class="s2">, </span><span class="s1">ctypes.byref(new_identity)</span>
372              <span class="s1">)</span>
373              <span class="s1">_assert_no_error(status)</span>
374              <span class="s1">identities.append(new_identity)</span>
375  
376              <span class="s3"># We now want to release the original certificate, as we no longer</span>
377              <span class="s3"># need it.</span>
378              <span class="s1">CoreFoundation.CFRelease(certificates.pop(</span><span class="s5">0</span><span class="s1">))</span>
379  
380          <span class="s3"># We now need to build a new CFArray that holds the trust chain.</span>
381          <span class="s1">trust_chain = CoreFoundation.CFArrayCreateMutable(</span>
382              <span class="s1">CoreFoundation.kCFAllocatorDefault</span><span class="s2">,</span>
383              <span class="s5">0</span><span class="s2">,</span>
384              <span class="s1">ctypes.byref(CoreFoundation.kCFTypeArrayCallBacks)</span><span class="s2">,</span>
385          <span class="s1">)</span>
386          <span class="s2">for </span><span class="s1">item </span><span class="s2">in </span><span class="s1">itertools.chain(identities</span><span class="s2">, </span><span class="s1">certificates):</span>
387              <span class="s3"># ArrayAppendValue does a CFRetain on the item. That's fine,</span>
388              <span class="s3"># because the finally block will release our other refs to them.</span>
389              <span class="s1">CoreFoundation.CFArrayAppendValue(trust_chain</span><span class="s2">, </span><span class="s1">item)</span>
390  
391          <span class="s2">return </span><span class="s1">trust_chain</span>
392      <span class="s2">finally</span><span class="s1">:</span>
393          <span class="s2">for </span><span class="s1">obj </span><span class="s2">in </span><span class="s1">itertools.chain(identities</span><span class="s2">, </span><span class="s1">certificates):</span>
394              <span class="s1">CoreFoundation.CFRelease(obj)</span>
