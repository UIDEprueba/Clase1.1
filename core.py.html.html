77   <span class="s2">def </span><span class="s1">setup (**attrs):</span>
78       <span class="s0">&quot;&quot;&quot;The gateway to the Distutils: do everything your setup script needs 
79       to do, in a highly flexible and user-driven way.  Briefly: create a 
80       Distribution instance; find and parse config files; parse the command 
81       line; run each Distutils command found there, customized by the options 
82       supplied to 'setup()' (as keyword arguments), in config files, and on 
83       the command line. 
84    
85       The Distribution instance might be an instance of a class supplied via 
86       the 'distclass' keyword argument to 'setup'; if no such class is 
87       supplied, then the Distribution class (in dist.py) is instantiated. 
88       All other arguments to 'setup' (except for 'cmdclass') are used to set 
89       attributes of the Distribution instance. 
90    
91       The 'cmdclass' argument, if supplied, is a dictionary mapping command 
92       names to command classes.  Each command encountered on the command line 
93       will be turned into a command class, which is in turn instantiated; any 
94       class found in 'cmdclass' is used in place of the default, which is 
95       (for command 'foo_bar') class 'foo_bar' in module 
96       'distutils.command.foo_bar'.  The command class must provide a 
97       'user_options' attribute which is a list of option specifiers for 
98       'distutils.fancy_getopt'.  Any command-line options between the current 
99       and the next command are used to set attributes of the current command 
100      object. 
101   
102      When the entire command-line has been successfully parsed, calls the 
103      'run()' method on each command object in turn.  This method will be 
104      driven entirely by the Distribution object (which each command object 
105      has a reference to, thanks to its constructor), and the 
106      command-specific options that became attributes of each command 
107      object. 
108      &quot;&quot;&quot;</span>
109  
110      <span class="s2">global </span><span class="s1">_setup_stop_after</span><span class="s2">, </span><span class="s1">_setup_distribution</span>
111  
112      <span class="s3"># Determine the distribution class -- either caller-supplied or</span>
113      <span class="s3"># our Distribution (see below).</span>
114      <span class="s1">klass = attrs.get(</span><span class="s4">'distclass'</span><span class="s1">)</span>
115      <span class="s2">if </span><span class="s1">klass:</span>
116          <span class="s2">del </span><span class="s1">attrs[</span><span class="s4">'distclass'</span><span class="s1">]</span>
117      <span class="s2">else</span><span class="s1">:</span>
118          <span class="s1">klass = Distribution</span>
119  
120      <span class="s2">if </span><span class="s4">'script_name' </span><span class="s2">not in </span><span class="s1">attrs:</span>
121          <span class="s1">attrs[</span><span class="s4">'script_name'</span><span class="s1">] = os.path.basename(sys.argv[</span><span class="s5">0</span><span class="s1">])</span>
122      <span class="s2">if </span><span class="s4">'script_args'  </span><span class="s2">not in </span><span class="s1">attrs:</span>
123          <span class="s1">attrs[</span><span class="s4">'script_args'</span><span class="s1">] = sys.argv[</span><span class="s5">1</span><span class="s1">:]</span>
124  
125      <span class="s3"># Create the Distribution instance, using the remaining arguments</span>
126      <span class="s3"># (ie. everything except distclass) to initialize it</span>
127      <span class="s2">try</span><span class="s1">:</span>
128          <span class="s1">_setup_distribution = dist = klass(attrs)</span>
129      <span class="s2">except </span><span class="s1">DistutilsSetupError </span><span class="s2">as </span><span class="s1">msg:</span>
130          <span class="s2">if </span><span class="s4">'name' </span><span class="s2">not in </span><span class="s1">attrs:</span>
131              <span class="s2">raise </span><span class="s1">SystemExit(</span><span class="s4">&quot;error in setup command: %s&quot; </span><span class="s1">% msg)</span>
132          <span class="s2">else</span><span class="s1">:</span>
133              <span class="s2">raise </span><span class="s1">SystemExit(</span><span class="s4">&quot;error in %s setup command: %s&quot; </span><span class="s1">% \</span>
134                    <span class="s1">(attrs[</span><span class="s4">'name'</span><span class="s1">]</span><span class="s2">, </span><span class="s1">msg))</span>
135  
136      <span class="s2">if </span><span class="s1">_setup_stop_after == </span><span class="s4">&quot;init&quot;</span><span class="s1">:</span>
137          <span class="s2">return </span><span class="s1">dist</span>
138  
139      <span class="s3"># Find and parse the config file(s): they will override options from</span>
140      <span class="s3"># the setup script, but be overridden by the command line.</span>
141      <span class="s1">dist.parse_config_files()</span>
142  
143      <span class="s2">if </span><span class="s1">DEBUG:</span>
144          <span class="s1">print(</span><span class="s4">&quot;options (after parsing config files):&quot;</span><span class="s1">)</span>
145          <span class="s1">dist.dump_option_dicts()</span>
146  
147      <span class="s2">if </span><span class="s1">_setup_stop_after == </span><span class="s4">&quot;config&quot;</span><span class="s1">:</span>
148          <span class="s2">return </span><span class="s1">dist</span>
149  
150      <span class="s3"># Parse the command line and override config files; any</span>
151      <span class="s3"># command-line errors are the end user's fault, so turn them into</span>
152      <span class="s3"># SystemExit to suppress tracebacks.</span>
153      <span class="s2">try</span><span class="s1">:</span>
154          <span class="s1">ok = dist.parse_command_line()</span>
155      <span class="s2">except </span><span class="s1">DistutilsArgError </span><span class="s2">as </span><span class="s1">msg:</span>
156          <span class="s2">raise </span><span class="s1">SystemExit(gen_usage(dist.script_name) + </span><span class="s4">&quot;</span><span class="s2">\n</span><span class="s4">error: %s&quot; </span><span class="s1">% msg)</span>
157  
158      <span class="s2">if </span><span class="s1">DEBUG:</span>
159          <span class="s1">print(</span><span class="s4">&quot;options (after parsing command line):&quot;</span><span class="s1">)</span>
160          <span class="s1">dist.dump_option_dicts()</span>
161  
162      <span class="s2">if </span><span class="s1">_setup_stop_after == </span><span class="s4">&quot;commandline&quot;</span><span class="s1">:</span>
163          <span class="s2">return </span><span class="s1">dist</span>
164  
165      <span class="s3"># And finally, run all the commands found on the command line.</span>
166      <span class="s2">if </span><span class="s1">ok:</span>
167          <span class="s2">return </span><span class="s1">run_commands(dist)</span>
168  
169      <span class="s2">return </span><span class="s1">dist</span>

175      <span class="s0">&quot;&quot;&quot;Given a Distribution object run all the commands, 
176      raising ``SystemExit`` errors in the case of failure. 
177   
178      This function assumes that either ``sys.argv`` or ``dist.script_args`` 
179      is already set accordingly. 
180      &quot;&quot;&quot;</span>
181      <span class="s2">try</span><span class="s1">:</span>
182          <span class="s1">dist.run_commands()</span>
183      <span class="s2">except </span><span class="s1">KeyboardInterrupt:</span>
184          <span class="s2">raise </span><span class="s1">SystemExit(</span><span class="s4">&quot;interrupted&quot;</span><span class="s1">)</span>
185      <span class="s2">except </span><span class="s1">OSError </span><span class="s2">as </span><span class="s1">exc:</span>
186          <span class="s2">if </span><span class="s1">DEBUG:</span>
187              <span class="s1">sys.stderr.write(</span><span class="s4">&quot;error: %s</span><span class="s2">\n</span><span class="s4">&quot; </span><span class="s1">% (exc</span><span class="s2">,</span><span class="s1">))</span>
188              <span class="s2">raise</span>
189          <span class="s2">else</span><span class="s1">:</span>
190              <span class="s2">raise </span><span class="s1">SystemExit(</span><span class="s4">&quot;error: %s&quot; </span><span class="s1">% (exc</span><span class="s2">,</span><span class="s1">))</span>
191  
192      <span class="s2">except </span><span class="s1">(DistutilsError</span><span class="s2">,</span>
193              <span class="s1">CCompilerError) </span><span class="s2">as </span><span class="s1">msg:</span>
194          <span class="s2">if </span><span class="s1">DEBUG:</span>
195              <span class="s2">raise</span>
196          <span class="s2">else</span><span class="s1">:</span>
197              <span class="s2">raise </span><span class="s1">SystemExit(</span><span class="s4">&quot;error: &quot; </span><span class="s1">+ str(msg))</span>

202  <span class="s2">def </span><span class="s1">run_setup (script_name</span><span class="s2">, </span><span class="s1">script_args=</span><span class="s2">None, </span><span class="s1">stop_after=</span><span class="s4">&quot;run&quot;</span><span class="s1">):</span>
203      <span class="s0">&quot;&quot;&quot;Run a setup script in a somewhat controlled environment, and 
204      return the Distribution instance that drives things.  This is useful 
205      if you need to find out the distribution meta-data (passed as 
206      keyword args from 'script' to 'setup()', or the contents of the 
207      config files or command-line. 
208   
209      'script_name' is a file that will be read and run with 'exec()'; 
210      'sys.argv[0]' will be replaced with 'script' for the duration of the 
211      call.  'script_args' is a list of strings; if supplied, 
212      'sys.argv[1:]' will be replaced by 'script_args' for the duration of 
213      the call. 
214   
215      'stop_after' tells 'setup()' when to stop processing; possible 
216      values: 
217        init 
218          stop after the Distribution instance has been created and 
219          populated with the keyword arguments to 'setup()' 
220        config 
221          stop after config files have been parsed (and their data 
222          stored in the Distribution instance) 
223        commandline 
224          stop after the command-line ('sys.argv[1:]' or 'script_args') 
225          have been parsed (and the data stored in the Distribution) 
226        run [default] 
227          stop after all commands have been run (the same as if 'setup()' 
228          had been called in the usual way 
229   
230      Returns the Distribution instance, which provides all information 
231      used to drive the Distutils. 
232      &quot;&quot;&quot;</span>
233      <span class="s2">if </span><span class="s1">stop_after </span><span class="s2">not in </span><span class="s1">(</span><span class="s4">'init'</span><span class="s2">, </span><span class="s4">'config'</span><span class="s2">, </span><span class="s4">'commandline'</span><span class="s2">, </span><span class="s4">'run'</span><span class="s1">):</span>
234          <span class="s2">raise </span><span class="s1">ValueError(</span><span class="s4">&quot;invalid value for 'stop_after': %r&quot; </span><span class="s1">% (stop_after</span><span class="s2">,</span><span class="s1">))</span>
235  
236      <span class="s2">global </span><span class="s1">_setup_stop_after</span><span class="s2">, </span><span class="s1">_setup_distribution</span>
237      <span class="s1">_setup_stop_after = stop_after</span>
238  
239      <span class="s1">save_argv = sys.argv.copy()</span>
240      <span class="s1">g = {</span><span class="s4">'__file__'</span><span class="s1">: script_name</span><span class="s2">, </span><span class="s4">'__name__'</span><span class="s1">: </span><span class="s4">'__main__'</span><span class="s1">}</span>
241      <span class="s2">try</span><span class="s1">:</span>
242          <span class="s2">try</span><span class="s1">:</span>
243              <span class="s1">sys.argv[</span><span class="s5">0</span><span class="s1">] = script_name</span>
244              <span class="s2">if </span><span class="s1">script_args </span><span class="s2">is not None</span><span class="s1">:</span>
245                  <span class="s1">sys.argv[</span><span class="s5">1</span><span class="s1">:] = script_args</span>
246              <span class="s3"># tokenize.open supports automatic encoding detection</span>
247              <span class="s2">with </span><span class="s1">tokenize.open(script_name) </span><span class="s2">as </span><span class="s1">f:</span>
248                  <span class="s1">code = f.read().replace(</span><span class="s4">r'\r\n'</span><span class="s2">, </span><span class="s4">r'\n'</span><span class="s1">)</span>
249                  <span class="s1">exec(code</span><span class="s2">, </span><span class="s1">g)</span>
250          <span class="s2">finally</span><span class="s1">:</span>
251              <span class="s1">sys.argv = save_argv</span>
252              <span class="s1">_setup_stop_after = </span><span class="s2">None</span>
253      <span class="s2">except </span><span class="s1">SystemExit:</span>
254          <span class="s3"># Hmm, should we do something if exiting with a non-zero code</span>
255          <span class="s3"># (ie. error)?</span>
256          <span class="s2">pass</span>
257  
258      <span class="s2">if </span><span class="s1">_setup_distribution </span><span class="s2">is None</span><span class="s1">:</span>
259          <span class="s2">raise </span><span class="s1">RuntimeError((</span><span class="s4">&quot;'distutils.core.setup()' was never called -- &quot;</span>
260                 <span class="s4">&quot;perhaps '%s' is not a Distutils setup script?&quot;</span><span class="s1">) % \</span>
261                <span class="s1">script_name)</span>
262  
263      <span class="s3"># I wonder if the setup script's namespace -- g and l -- would be of</span>
264      <span class="s3"># any interest to callers?</span>
265      <span class="s3">#print &quot;_setup_distribution:&quot;, _setup_distribution</span>
266      <span class="s2">return </span><span class="s1">_setup_distribution</span>
