459      <span class="s2">def </span><span class="s1">parse_command_line(self):</span>

465          and options for that command.  Each new command terminates the 
466          options for the previous command.  The allowed options for a 
467          command are determined by the 'user_options' attribute of the 
468          command class -- thus, we have to be able to load command classes 
469          in order to parse the command line.  Any error in that 'options' 
470          attribute raises DistutilsGetoptError; any error on the 
471          command-line raises DistutilsArgError.  If no Distutils commands 
472          were found on the command line, raises DistutilsArgError.  Return

481          <span class="s3">#</span>

483

485          <span class="s3"># options, then the first command, then its options, and so on --</span>
486          <span class="s3"># because each command will be handled by a different class, and</span>
487          <span class="s3"># the options that are valid for a particular class aren't known</span>
488          <span class="s3"># until we have loaded the command class, which doesn't happen</span>
489          <span class="s3"># until we know what the command is.</span>

500          <span class="s2">if </span><span class="s1">self.handle_display_options(option_order):</span>

509          <span class="s3"># former, we show global options (--verbose, --dry-run, etc.)</span>
510          <span class="s3"># and display-only options (--name, --version, etc.); for the</span>
511          <span class="s3"># latter, we omit the display-only options and show help for</span>
512          <span class="s3"># each command listed on the command line.</span>
513          <span class="s2">if </span><span class="s1">self.help:</span>

520          <span class="s2">if not </span><span class="s1">self.commands:</span>
521              <span class="s2">raise </span><span class="s1">DistutilsArgError(</span><span class="s4">&quot;no commands supplied&quot;</span><span class="s1">)</span>
522  
523          <span class="s3"># All is well: return true</span>
524          <span class="s2">return True</span>
