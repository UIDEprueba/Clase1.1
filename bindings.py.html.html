84   <span class="s2">def </span><span class="s1">load_cdll(name</span><span class="s2">, </span><span class="s1">macos10_16_path):</span>
85       <span class="s0">&quot;&quot;&quot;Loads a CDLL by name, falling back to known path on 10.16+&quot;&quot;&quot;</span>
86       <span class="s2">try</span><span class="s1">:</span>
87           <span class="s5"># Big Sur is technically 11 but we use 10.16 due to the Big Sur</span>
88           <span class="s5"># beta being labeled as 10.16.</span>
89           <span class="s2">if </span><span class="s1">version_info &gt;= (</span><span class="s4">10</span><span class="s2">, </span><span class="s4">16</span><span class="s1">):</span>
90               <span class="s1">path = macos10_16_path</span>
91           <span class="s2">else</span><span class="s1">:</span>
92               <span class="s1">path = find_library(name)</span>
93           <span class="s2">if not </span><span class="s1">path:</span>
94               <span class="s2">raise </span><span class="s1">OSError  </span><span class="s5"># Caught and reraised as 'ImportError'</span>
95           <span class="s2">return </span><span class="s1">CDLL(path</span><span class="s2">, </span><span class="s1">use_errno=</span><span class="s2">True</span><span class="s1">)</span>
96       <span class="s2">except </span><span class="s1">OSError:</span>
97           <span class="s1">raise_from(ImportError(</span><span class="s3">&quot;The library %s failed to load&quot; </span><span class="s1">% name)</span><span class="s2">, None</span><span class="s1">)</span>
