201      <span class="s2">def </span><span class="s1">include_pattern(self</span><span class="s2">, </span><span class="s1">pattern</span><span class="s2">, </span><span class="s1">anchor=</span><span class="s4">1</span><span class="s2">, </span><span class="s1">prefix=</span><span class="s2">None, </span><span class="s1">is_regex=</span><span class="s4">0</span><span class="s1">):</span>
202          <span class="s0">&quot;&quot;&quot;Select strings (presumably filenames) from 'self.files' that 
203          match 'pattern', a Unix-style wildcard (glob) pattern.  Patterns 
204          are not quite the same as implemented by the 'fnmatch' module: '*' 
205          and '?'  match non-special characters, where &quot;special&quot; is platform- 
206          dependent: slash on Unix; colon, slash, and backslash on 
207          DOS/Windows; and colon on Mac OS. 
208   
209          If 'anchor' is true (the default), then the pattern match is more 
210          stringent: &quot;*.py&quot; will match &quot;foo.py&quot; but not &quot;foo/bar.py&quot;.  If 
211          'anchor' is false, both of these will match. 
212

214          (itself a pattern) and ending with 'pattern', with anything in between 
215          them, will match.  'anchor' is ignored in this case. 
216   
217          If 'is_regex' is true, 'anchor' and 'prefix' are ignored, and 
218          'pattern' is assumed to be either a string containing a regex or a 
219          regex object -- no translation is done, the regex is just compiled 
220          and used as-is. 
221   
222          Selected strings will be added to self.files. 
223   
224          Return True if files are found, False otherwise. 
225          &quot;&quot;&quot;</span>

316  <span class="s2">def </span><span class="s1">glob_to_re(pattern):</span>
317      <span class="s0">&quot;&quot;&quot;Translate a shell-like glob pattern to a regular expression; return

319      that '*' does not match &quot;special characters&quot; (which are

326      <span class="s3"># and by extension they shouldn't match such &quot;special characters&quot; under</span>
327      <span class="s3"># any OS.  So change all non-escaped dots in the RE to match any</span>

329      <span class="s1">sep = os.sep</span>

332          <span class="s3"># to escape the backslash twice</span>

334      <span class="s1">escaped = </span><span class="s5">r'\1[^%s]' </span><span class="s1">% sep</span>
335      <span class="s1">pattern_re = re.sub(</span><span class="s5">r'((?&lt;!\\)(\\\\)*)\.'</span><span class="s2">, </span><span class="s1">escaped</span><span class="s2">, </span><span class="s1">pattern_re)</span>
336      <span class="s2">return </span><span class="s1">pattern_re</span>
