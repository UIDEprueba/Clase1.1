28   <span class="s3"># This class is really only used by the &quot;build_ext&quot; command, so it might</span>
29   <span class="s3"># make sense to put it in distutils.command.build_ext.  However, that</span>
30   <span class="s3"># module is already big enough, and I want to make this class a bit more</span>
31   <span class="s3"># complex to simplify some common cases (&quot;foo&quot; module in &quot;foo.c&quot;) and do</span>

38   <span class="s2">class </span><span class="s1">Extension:</span>
39       <span class="s0">&quot;&quot;&quot;Just a collection of attributes that describes an extension 
40       module and everything needed to build it (hopefully in a portable 
41       way, but there are hooks that let you be as unportable as you need). 
42    
43       Instance attributes: 
44         name : string 
45           the full name of the extension, including any packages -- ie. 
46           *not* a filename or pathname, but Python dotted name 
47         sources : [string] 
48           list of source filenames, relative to the distribution root 
49           (where the setup script lives), in Unix form (slash-separated) 
50           for portability.  Source files may be C, C++, SWIG (.i), 
51           platform-specific resource files, or whatever else is recognized 
52           by the &quot;build_ext&quot; command as source for a Python extension. 
53         include_dirs : [string] 
54           list of directories to search for C/C++ header files (in Unix 
55           form for portability) 
56         define_macros : [(name : string, value : string|None)] 
57           list of macros to define; each macro is defined using a 2-tuple, 
58           where 'value' is either the string to define it to or None to 
59           define it without a particular value (equivalent of &quot;#define 
60           FOO&quot; in source or -DFOO on Unix C compiler command line) 
61         undef_macros : [string] 
62           list of macros to undefine explicitly 
63         library_dirs : [string] 
64           list of directories to search for C/C++ libraries at link time 
65         libraries : [string] 
66           list of library names (not filenames or paths) to link against 
67         runtime_library_dirs : [string] 
68           list of directories to search for C/C++ libraries at run time 
69           (for shared extensions, this is when the extension is loaded) 
70         extra_objects : [string] 
71           list of extra files to link with (eg. object files not implied 
72           by 'sources', static library that must be explicitly specified, 
73           binary resource files, etc.) 
74         extra_compile_args : [string] 
75           any extra platform- and compiler-specific information to use 
76           when compiling the source files in 'sources'.  For platforms and 
77           compilers where &quot;command line&quot; makes sense, this is typically a 
78           list of command-line arguments, but for other platforms it could 
79           be anything. 
80         extra_link_args : [string] 
81           any extra platform- and compiler-specific information to use 
82           when linking object files together to create the extension (or 
83           to create a new static Python interpreter).  Similar 
84           interpretation as for 'extra_compile_args'. 
85         export_symbols : [string] 
86           list of symbols to be exported from a shared extension.  Not 
87           used on all platforms, and not generally necessary for Python 
88           extensions, which typically export exactly one symbol: &quot;init&quot; + 
89           extension_name. 
90         swig_opts : [string] 
91           any extra options to pass to SWIG if a source file has the .i 
92           extension. 
93         depends : [string] 
94           list of files that the extension depends on 
95         language : string 
96           extension language (i.e. &quot;c&quot;, &quot;c++&quot;, &quot;objc&quot;). Will be detected 
97           from the source extensions if not provided. 
98         optional : boolean 
99           specifies that a build failure in the extension should not abort the 
100          build process, but simply not install the failing extension. 
101      &quot;&quot;&quot;</span>

160  <span class="s2">def </span><span class="s1">read_setup_file(filename):</span>
161      <span class="s0">&quot;&quot;&quot;Reads a Setup file and returns Extension instances.&quot;&quot;&quot;</span>
162      <span class="s2">from </span><span class="s1">distutils.sysconfig </span><span class="s2">import </span><span class="s1">(parse_makefile</span><span class="s2">, </span><span class="s1">expand_makefile_vars</span><span class="s2">,</span>
163                                       <span class="s1">_variable_rx)</span>
164  
165      <span class="s2">from </span><span class="s1">distutils.text_file </span><span class="s2">import </span><span class="s1">TextFile</span>
166      <span class="s2">from </span><span class="s1">distutils.util </span><span class="s2">import </span><span class="s1">split_quoted</span>
167  
168      <span class="s3"># First pass over the file to gather &quot;VAR = VALUE&quot; assignments.</span>
169      <span class="s1">vars = parse_makefile(filename)</span>
170  
171      <span class="s3"># Second pass to gobble up the real content: lines of the form</span>
172      <span class="s3">#   &lt;module&gt; ... [&lt;sourcefile&gt; ...] [&lt;cpparg&gt; ...] [&lt;library&gt; ...]</span>
173      <span class="s1">file = TextFile(filename</span><span class="s2">,</span>
174                      <span class="s1">strip_comments=</span><span class="s5">1</span><span class="s2">, </span><span class="s1">skip_blanks=</span><span class="s5">1</span><span class="s2">, </span><span class="s1">join_lines=</span><span class="s5">1</span><span class="s2">,</span>
175                      <span class="s1">lstrip_ws=</span><span class="s5">1</span><span class="s2">, </span><span class="s1">rstrip_ws=</span><span class="s5">1</span><span class="s1">)</span>
176      <span class="s2">try</span><span class="s1">:</span>
177          <span class="s1">extensions = []</span>
178  
179          <span class="s2">while True</span><span class="s1">:</span>
180              <span class="s1">line = file.readline()</span>
181              <span class="s2">if </span><span class="s1">line </span><span class="s2">is None</span><span class="s1">:                </span><span class="s3"># eof</span>
182                  <span class="s2">break</span>
183              <span class="s2">if </span><span class="s1">_variable_rx.match(line):    </span><span class="s3"># VAR=VALUE, handled in first pass</span>
184                  <span class="s2">continue</span>
185  
186              <span class="s2">if </span><span class="s1">line[</span><span class="s5">0</span><span class="s1">] == line[-</span><span class="s5">1</span><span class="s1">] == </span><span class="s4">&quot;*&quot;</span><span class="s1">:</span>
187                  <span class="s1">file.warn(</span><span class="s4">&quot;'%s' lines not handled yet&quot; </span><span class="s1">% line)</span>
188                  <span class="s2">continue</span>
189  
190              <span class="s1">line = expand_makefile_vars(line</span><span class="s2">, </span><span class="s1">vars)</span>
191              <span class="s1">words = split_quoted(line)</span>
192  
193              <span class="s3"># NB. this parses a slightly different syntax than the old</span>
194              <span class="s3"># makesetup script: here, there must be exactly one extension per</span>
195              <span class="s3"># line, and it must be the first word of the line.  I have no idea</span>
196              <span class="s3"># why the old syntax supported multiple extensions per line, as</span>
197              <span class="s3"># they all wind up being the same.</span>
198  
199              <span class="s1">module = words[</span><span class="s5">0</span><span class="s1">]</span>
200              <span class="s1">ext = Extension(module</span><span class="s2">, </span><span class="s1">[])</span>
201              <span class="s1">append_next_word = </span><span class="s2">None</span>
202  
203              <span class="s2">for </span><span class="s1">word </span><span class="s2">in </span><span class="s1">words[</span><span class="s5">1</span><span class="s1">:]:</span>
204                  <span class="s2">if </span><span class="s1">append_next_word </span><span class="s2">is not None</span><span class="s1">:</span>
205                      <span class="s1">append_next_word.append(word)</span>
206                      <span class="s1">append_next_word = </span><span class="s2">None</span>
207                      <span class="s2">continue</span>
208  
209                  <span class="s1">suffix = os.path.splitext(word)[</span><span class="s5">1</span><span class="s1">]</span>
210                  <span class="s1">switch = word[</span><span class="s5">0</span><span class="s1">:</span><span class="s5">2</span><span class="s1">] ; value = word[</span><span class="s5">2</span><span class="s1">:]</span>
211  
212                  <span class="s2">if </span><span class="s1">suffix </span><span class="s2">in </span><span class="s1">(</span><span class="s4">&quot;.c&quot;</span><span class="s2">, </span><span class="s4">&quot;.cc&quot;</span><span class="s2">, </span><span class="s4">&quot;.cpp&quot;</span><span class="s2">, </span><span class="s4">&quot;.cxx&quot;</span><span class="s2">, </span><span class="s4">&quot;.c++&quot;</span><span class="s2">, </span><span class="s4">&quot;.m&quot;</span><span class="s2">, </span><span class="s4">&quot;.mm&quot;</span><span class="s1">):</span>
213                      <span class="s3"># hmm, should we do something about C vs. C++ sources?</span>
214                      <span class="s3"># or leave it up to the CCompiler implementation to</span>
215                      <span class="s3"># worry about?</span>
216                      <span class="s1">ext.sources.append(word)</span>
217                  <span class="s2">elif </span><span class="s1">switch == </span><span class="s4">&quot;-I&quot;</span><span class="s1">:</span>
218                      <span class="s1">ext.include_dirs.append(value)</span>
219                  <span class="s2">elif </span><span class="s1">switch == </span><span class="s4">&quot;-D&quot;</span><span class="s1">:</span>
220                      <span class="s1">equals = value.find(</span><span class="s4">&quot;=&quot;</span><span class="s1">)</span>
221                      <span class="s2">if </span><span class="s1">equals == -</span><span class="s5">1</span><span class="s1">:        </span><span class="s3"># bare &quot;-DFOO&quot; -- no value</span>
222                          <span class="s1">ext.define_macros.append((value</span><span class="s2">, None</span><span class="s1">))</span>
223                      <span class="s2">else</span><span class="s1">:                   </span><span class="s3"># &quot;-DFOO=blah&quot;</span>
224                          <span class="s1">ext.define_macros.append((value[</span><span class="s5">0</span><span class="s1">:equals]</span><span class="s2">,</span>
225                                                    <span class="s1">value[equals+</span><span class="s5">2</span><span class="s1">:]))</span>
226                  <span class="s2">elif </span><span class="s1">switch == </span><span class="s4">&quot;-U&quot;</span><span class="s1">:</span>
227                      <span class="s1">ext.undef_macros.append(value)</span>
228                  <span class="s2">elif </span><span class="s1">switch == </span><span class="s4">&quot;-C&quot;</span><span class="s1">:        </span><span class="s3"># only here 'cause makesetup has it!</span>
229                      <span class="s1">ext.extra_compile_args.append(word)</span>
230                  <span class="s2">elif </span><span class="s1">switch == </span><span class="s4">&quot;-l&quot;</span><span class="s1">:</span>
231                      <span class="s1">ext.libraries.append(value)</span>
232                  <span class="s2">elif </span><span class="s1">switch == </span><span class="s4">&quot;-L&quot;</span><span class="s1">:</span>
233                      <span class="s1">ext.library_dirs.append(value)</span>
234                  <span class="s2">elif </span><span class="s1">switch == </span><span class="s4">&quot;-R&quot;</span><span class="s1">:</span>
235                      <span class="s1">ext.runtime_library_dirs.append(value)</span>
236                  <span class="s2">elif </span><span class="s1">word == </span><span class="s4">&quot;-rpath&quot;</span><span class="s1">:</span>
237                      <span class="s1">append_next_word = ext.runtime_library_dirs</span>
238                  <span class="s2">elif </span><span class="s1">word == </span><span class="s4">&quot;-Xlinker&quot;</span><span class="s1">:</span>
239                      <span class="s1">append_next_word = ext.extra_link_args</span>
240                  <span class="s2">elif </span><span class="s1">word == </span><span class="s4">&quot;-Xcompiler&quot;</span><span class="s1">:</span>
241                      <span class="s1">append_next_word = ext.extra_compile_args</span>
242                  <span class="s2">elif </span><span class="s1">switch == </span><span class="s4">&quot;-u&quot;</span><span class="s1">:</span>
243                      <span class="s1">ext.extra_link_args.append(word)</span>
244                      <span class="s2">if not </span><span class="s1">value:</span>
245                          <span class="s1">append_next_word = ext.extra_link_args</span>
246                  <span class="s2">elif </span><span class="s1">suffix </span><span class="s2">in </span><span class="s1">(</span><span class="s4">&quot;.a&quot;</span><span class="s2">, </span><span class="s4">&quot;.so&quot;</span><span class="s2">, </span><span class="s4">&quot;.sl&quot;</span><span class="s2">, </span><span class="s4">&quot;.o&quot;</span><span class="s2">, </span><span class="s4">&quot;.dylib&quot;</span><span class="s1">):</span>
247                      <span class="s3"># NB. a really faithful emulation of makesetup would</span>
248                      <span class="s3"># append a .o file to extra_objects only if it</span>
249                      <span class="s3"># had a slash in it; otherwise, it would s/.o/.c/</span>
250                      <span class="s3"># and append it to sources.  Hmmmm.</span>
251                      <span class="s1">ext.extra_objects.append(word)</span>
252                  <span class="s2">else</span><span class="s1">:</span>
253                      <span class="s1">file.warn(</span><span class="s4">&quot;unrecognized argument '%s'&quot; </span><span class="s1">% word)</span>
254  
255              <span class="s1">extensions.append(ext)</span>
256      <span class="s2">finally</span><span class="s1">:</span>
257          <span class="s1">file.close()</span>
258  
259      <span class="s2">return </span><span class="s1">extensions</span>
