45       <span class="s4">def </span><span class="s1">__init__(self):</span>
46           <span class="s0"># the subclasses declare class attributes which are numbers.</span>
47           <span class="s0"># Upon instantiation we define instance attributes, which are the same</span>

50               <span class="s4">if not </span><span class="s1">name.startswith(</span><span class="s3">'_'</span><span class="s1">):</span>
51                   <span class="s1">value = getattr(self</span><span class="s4">, </span><span class="s1">name)</span>
