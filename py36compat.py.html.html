34       <span class="s0">def </span><span class="s1">add_defaults(self):</span>
35           <span class="s2">&quot;&quot;&quot;Add all the default files to self.filelist: 
36             - README or README.txt 
37             - setup.py 
38             - test/test*.py 
39             - all pure Python modules mentioned in setup script 
40             - all files pointed by package_data (build_py) 
41             - all files defined in data_files. 
42             - all files defined as scripts. 
43             - all C sources listed as part of extensions or C libraries 
44               in the setup script (doesn't catch C headers!) 
45           Warns if (README or README.txt) or setup.py are missing; everything 
46           else is optional.

58           <span class="s2">&quot;&quot;&quot;

61           &gt;&gt;&gt; sdist_add_defaults._cs_path_exists(__file__)

67               <span class="s0">return False</span>

69           <span class="s1">abspath = os.path.abspath(fspath)</span>
70           <span class="s1">directory</span><span class="s0">, </span><span class="s1">filename = os.path.split(abspath)</span>
71           <span class="s0">return </span><span class="s1">filename </span><span class="s0">in </span><span class="s1">os.listdir(directory)</span>
