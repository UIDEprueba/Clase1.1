49   <span class="s0">def </span><span class="s1">_should_use_osx_framework_prefix() -&gt; bool:</span>
50       <span class="s4">&quot;&quot;&quot;Check for Apple's ``osx_framework_library`` scheme. 
51    
52       Python distributed by Apple's Command Line Tools has this special scheme 
53       that's used when: 
54    
55       * This is a framework build. 
56       * We are installing into the system prefix. 
57    
58       This does not account for ``pip install --prefix`` (also means we're not 
59       installing to the system prefix), which should use ``posix_prefix``, but 
60       logic here means ``_infer_prefix()`` outputs ``osx_framework_library``. But 
61       since ``prefix`` is not available for ``sysconfig.get_default_scheme()``, 
62       which is the stdlib replacement for ``_infer_prefix()``, presumably Apple 
63       wouldn't be able to magically switch between ``osx_framework_library`` and 
64       ``posix_prefix``. ``_infer_prefix()`` returning ``osx_framework_library`` 
65       means its behavior is consistent whether we use the stdlib implementation 
66       or our own, and we deal with this special case in ``get_scheme()`` instead. 
67       &quot;&quot;&quot;</span>
68       <span class="s0">return </span><span class="s1">(</span>
69           <span class="s3">&quot;osx_framework_library&quot; </span><span class="s0">in </span><span class="s1">_AVAILABLE_SCHEMES</span>
70           <span class="s0">and not </span><span class="s1">running_under_virtualenv()</span>
71           <span class="s0">and </span><span class="s1">is_osx_framework()</span>
72       <span class="s1">)</span>

75   <span class="s0">def </span><span class="s1">_infer_prefix() -&gt; str:</span>
76       <span class="s4">&quot;&quot;&quot;Try to find a prefix scheme for the current platform. 
77    
78       This tries: 
79    
80       * A special ``osx_framework_library`` for Python distributed by Apple's 
81         Command Line Tools, when not running in a virtual environment. 
82       * Implementation + OS, used by PyPy on Windows (``pypy_nt``). 
83       * Implementation without OS, used by PyPy on POSIX (``pypy``). 
84       * OS + &quot;prefix&quot;, used by CPython on POSIX (``posix_prefix``). 
85       * Just the OS name, used by CPython on Windows (``nt``). 
86    
87       If none of the above works, fall back to ``posix_prefix``. 
88       &quot;&quot;&quot;</span>
89       <span class="s0">if </span><span class="s1">_PREFERRED_SCHEME_API:</span>
90           <span class="s0">return </span><span class="s1">_PREFERRED_SCHEME_API(</span><span class="s3">&quot;prefix&quot;</span><span class="s1">)</span>
91       <span class="s0">if </span><span class="s1">_should_use_osx_framework_prefix():</span>
92           <span class="s0">return </span><span class="s3">&quot;osx_framework_library&quot;</span>
93       <span class="s1">implementation_suffixed = </span><span class="s3">f&quot;</span><span class="s0">{</span><span class="s1">sys.implementation.name</span><span class="s0">}</span><span class="s3">_</span><span class="s0">{</span><span class="s1">os.name</span><span class="s0">}</span><span class="s3">&quot;</span>
94       <span class="s0">if </span><span class="s1">implementation_suffixed </span><span class="s0">in </span><span class="s1">_AVAILABLE_SCHEMES:</span>
95           <span class="s0">return </span><span class="s1">implementation_suffixed</span>
96       <span class="s0">if </span><span class="s1">sys.implementation.name </span><span class="s0">in </span><span class="s1">_AVAILABLE_SCHEMES:</span>
97           <span class="s0">return </span><span class="s1">sys.implementation.name</span>
98       <span class="s1">suffixed = </span><span class="s3">f&quot;</span><span class="s0">{</span><span class="s1">os.name</span><span class="s0">}</span><span class="s3">_prefix&quot;</span>
99       <span class="s0">if </span><span class="s1">suffixed </span><span class="s0">in </span><span class="s1">_AVAILABLE_SCHEMES:</span>
100          <span class="s0">return </span><span class="s1">suffixed</span>
101      <span class="s0">if </span><span class="s1">os.name </span><span class="s0">in </span><span class="s1">_AVAILABLE_SCHEMES:  </span><span class="s2"># On Windows, prefx is just called &quot;nt&quot;.</span>
