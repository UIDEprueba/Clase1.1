72       <span class="s2">&quot;&quot;&quot; 
73       The ``UniversalDetector`` class underlies the ``chardet.detect`` function 
74       and coordinates all of the different charset probers. 
75

77       run: 
78

82               u.feed(some_bytes)

84               detected = u.result

240      <span class="s3">def </span><span class="s1">close(self):</span>
241          <span class="s2">&quot;&quot;&quot; 
242          Stop analyzing the current document and come up with a final 
243          prediction. 
244   
245          :returns:  The ``result`` attribute, a ``dict`` with the keys 
246                     `encoding`, `confidence`, and `language`. 
247          &quot;&quot;&quot;</span>
248          <span class="s0"># Don't bother with checks if we're already done</span>
249          <span class="s3">if </span><span class="s1">self.done:</span>
250              <span class="s3">return </span><span class="s1">self.result</span>
251          <span class="s1">self.done = </span><span class="s3">True</span>
252  
253          <span class="s3">if not </span><span class="s1">self._got_data:</span>

256          <span class="s0"># Default to ASCII if it is all we've seen so far</span>
257          <span class="s3">elif </span><span class="s1">self._input_state == InputState.PURE_ASCII:</span>
258              <span class="s1">self.result = {</span><span class="s6">'encoding'</span><span class="s1">: </span><span class="s6">'ascii'</span><span class="s3">,</span>
259                             <span class="s6">'confidence'</span><span class="s1">: </span><span class="s4">1.0</span><span class="s3">,</span>
260                             <span class="s6">'language'</span><span class="s1">: </span><span class="s6">''</span><span class="s1">}</span>
261  
262          <span class="s0"># If we have seen non-ASCII, return the best that met MINIMUM_THRESHOLD</span>
263          <span class="s3">elif </span><span class="s1">self._input_state == InputState.HIGH_BYTE:</span>
264              <span class="s1">prober_confidence = </span><span class="s3">None</span>
265              <span class="s1">max_prober_confidence = </span><span class="s4">0.0</span>
266              <span class="s1">max_prober = </span><span class="s3">None</span>
267              <span class="s3">for </span><span class="s1">prober </span><span class="s3">in </span><span class="s1">self._charset_probers:</span>
268                  <span class="s3">if not </span><span class="s1">prober:</span>
269                      <span class="s3">continue</span>
270                  <span class="s1">prober_confidence = prober.get_confidence()</span>
271                  <span class="s3">if </span><span class="s1">prober_confidence &gt; max_prober_confidence:</span>
272                      <span class="s1">max_prober_confidence = prober_confidence</span>
273                      <span class="s1">max_prober = prober</span>

276                  <span class="s1">lower_charset_name = max_prober.charset_name.lower()</span>
277                  <span class="s1">confidence = max_prober.get_confidence()</span>
278                  <span class="s0"># Use Windows encoding name instead of ISO-8859 if we saw any</span>
279                  <span class="s0"># extra Windows-specific bytes</span>
