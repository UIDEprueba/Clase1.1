44   <span class="s3">def </span><span class="s1">_dnsname_match(dn</span><span class="s3">, </span><span class="s1">hostname</span><span class="s3">, </span><span class="s1">max_wildcards=</span><span class="s5">1</span><span class="s1">):</span>
45       <span class="s0">&quot;&quot;&quot;Matching according to RFC 6125, section 6.4.3 
46    
47       http://tools.ietf.org/html/rfc6125#section-6.4.3 
48       &quot;&quot;&quot;</span>
49       <span class="s1">pats = []</span>
50       <span class="s3">if not </span><span class="s1">dn:</span>
51           <span class="s3">return False</span>
52   
53       <span class="s2"># Ported from python3-syntax:</span>
54       <span class="s2"># leftmost, *remainder = dn.split(r'.')</span>
55       <span class="s1">parts = dn.split(</span><span class="s4">r&quot;.&quot;</span><span class="s1">)</span>
56       <span class="s1">leftmost = parts[</span><span class="s5">0</span><span class="s1">]</span>
57       <span class="s1">remainder = parts[</span><span class="s5">1</span><span class="s1">:]</span>
58   
59       <span class="s1">wildcards = leftmost.count(</span><span class="s4">&quot;*&quot;</span><span class="s1">)</span>
60       <span class="s3">if </span><span class="s1">wildcards &gt; max_wildcards:</span>
61           <span class="s2"># Issue #17980: avoid denials of service by refusing more</span>
62           <span class="s2"># than one wildcard per fragment.  A survey of established</span>
63           <span class="s2"># policy among SSL implementations showed it to be a</span>
64           <span class="s2"># reasonable choice.</span>
65           <span class="s3">raise </span><span class="s1">CertificateError(</span>
66               <span class="s4">&quot;too many wildcards in certificate DNS name: &quot; </span><span class="s1">+ repr(dn)</span>
67           <span class="s1">)</span>
68   
69       <span class="s2"># speed up common case w/o wildcards</span>
70       <span class="s3">if not </span><span class="s1">wildcards:</span>
71           <span class="s3">return </span><span class="s1">dn.lower() == hostname.lower()</span>
72   
73       <span class="s2"># RFC 6125, section 6.4.3, subitem 1.</span>
74       <span class="s2"># The client SHOULD NOT attempt to match a presented identifier in which</span>
75       <span class="s2"># the wildcard character comprises a label other than the left-most label.</span>
76       <span class="s3">if </span><span class="s1">leftmost == </span><span class="s4">&quot;*&quot;</span><span class="s1">:</span>
77           <span class="s2"># When '*' is a fragment by itself, it matches a non-empty dotless</span>
78           <span class="s2"># fragment.</span>
79           <span class="s1">pats.append(</span><span class="s4">&quot;[^.]+&quot;</span><span class="s1">)</span>
80       <span class="s3">elif </span><span class="s1">leftmost.startswith(</span><span class="s4">&quot;xn--&quot;</span><span class="s1">) </span><span class="s3">or </span><span class="s1">hostname.startswith(</span><span class="s4">&quot;xn--&quot;</span><span class="s1">):</span>
81           <span class="s2"># RFC 6125, section 6.4.3, subitem 3.</span>
82           <span class="s2"># The client SHOULD NOT attempt to match a presented identifier</span>
83           <span class="s2"># where the wildcard character is embedded within an A-label or</span>
84           <span class="s2"># U-label of an internationalized domain name.</span>
85           <span class="s1">pats.append(re.escape(leftmost))</span>
86       <span class="s3">else</span><span class="s1">:</span>
87           <span class="s2"># Otherwise, '*' matches any dotless string, e.g. www*</span>
88           <span class="s1">pats.append(re.escape(leftmost).replace(</span><span class="s4">r&quot;\*&quot;</span><span class="s3">, </span><span class="s4">&quot;[^.]*&quot;</span><span class="s1">))</span>
89   
90       <span class="s2"># add the remaining fragments, ignore any wildcards</span>
91       <span class="s3">for </span><span class="s1">frag </span><span class="s3">in </span><span class="s1">remainder:</span>
92           <span class="s1">pats.append(re.escape(frag))</span>
93   
94       <span class="s1">pat = re.compile(</span><span class="s4">r&quot;\A&quot; </span><span class="s1">+ </span><span class="s4">r&quot;\.&quot;</span><span class="s1">.join(pats) + </span><span class="s4">r&quot;\Z&quot;</span><span class="s3">, </span><span class="s1">re.IGNORECASE)</span>
95       <span class="s3">return </span><span class="s1">pat.match(hostname)</span>

105  <span class="s3">def </span><span class="s1">_ipaddress_match(ipname</span><span class="s3">, </span><span class="s1">host_ip):</span>
106      <span class="s0">&quot;&quot;&quot;Exact matching of IP addresses. 
107   
108      RFC 6125 explicitly doesn't define an algorithm for this 
109      (section 1.7.2 - &quot;Out of Scope&quot;). 
110      &quot;&quot;&quot;</span>
111      <span class="s2"># OpenSSL may add a trailing newline to a subjectAltName's IP address</span>
112      <span class="s2"># Divergence from upstream: ipaddress can't handle byte str</span>
113      <span class="s1">ip = ipaddress.ip_address(_to_unicode(ipname).rstrip())</span>
114      <span class="s3">return </span><span class="s1">ip == host_ip</span>

117  <span class="s3">def </span><span class="s1">match_hostname(cert</span><span class="s3">, </span><span class="s1">hostname):</span>
118      <span class="s0">&quot;&quot;&quot;Verify that *cert* (in decoded format as returned by 
119      SSLSocket.getpeercert()) matches the *hostname*.  RFC 2818 and RFC 6125 
120      rules are followed, but IP addresses are not accepted for *hostname*. 
121   
122      CertificateError is raised on failure. On success, the function 
123      returns nothing. 
124      &quot;&quot;&quot;</span>
125      <span class="s3">if not </span><span class="s1">cert:</span>
126          <span class="s3">raise </span><span class="s1">ValueError(</span>
127              <span class="s4">&quot;empty or no certificate, match_hostname needs a &quot;</span>
128              <span class="s4">&quot;SSL socket or SSL context with either &quot;</span>
129              <span class="s4">&quot;CERT_OPTIONAL or CERT_REQUIRED&quot;</span>
130          <span class="s1">)</span>
131      <span class="s3">try</span><span class="s1">:</span>
132          <span class="s2"># Divergence from upstream: ipaddress can't handle byte str</span>
133          <span class="s1">host_ip = ipaddress.ip_address(_to_unicode(hostname))</span>
134      <span class="s3">except </span><span class="s1">(UnicodeError</span><span class="s3">, </span><span class="s1">ValueError):</span>
135          <span class="s2"># ValueError: Not an IP address (common case)</span>
136          <span class="s2"># UnicodeError: Divergence from upstream: Have to deal with ipaddress not taking</span>
137          <span class="s2"># byte strings.  addresses should be all ascii, so we consider it not</span>
138          <span class="s2"># an ipaddress in this case</span>
139          <span class="s1">host_ip = </span><span class="s3">None</span>
140      <span class="s3">except </span><span class="s1">AttributeError:</span>
141          <span class="s2"># Divergence from upstream: Make ipaddress library optional</span>
142          <span class="s3">if </span><span class="s1">ipaddress </span><span class="s3">is None</span><span class="s1">:</span>
143              <span class="s1">host_ip = </span><span class="s3">None</span>
144          <span class="s3">else</span><span class="s1">:  </span><span class="s2"># Defensive</span>
145              <span class="s3">raise</span>
146      <span class="s1">dnsnames = []</span>
147      <span class="s1">san = cert.get(</span><span class="s4">&quot;subjectAltName&quot;</span><span class="s3">, </span><span class="s1">())</span>
148      <span class="s3">for </span><span class="s1">key</span><span class="s3">, </span><span class="s1">value </span><span class="s3">in </span><span class="s1">san:</span>
149          <span class="s3">if </span><span class="s1">key == </span><span class="s4">&quot;DNS&quot;</span><span class="s1">:</span>
150              <span class="s3">if </span><span class="s1">host_ip </span><span class="s3">is None and </span><span class="s1">_dnsname_match(value</span><span class="s3">, </span><span class="s1">hostname):</span>
151                  <span class="s3">return</span>
152              <span class="s1">dnsnames.append(value)</span>
153          <span class="s3">elif </span><span class="s1">key == </span><span class="s4">&quot;IP Address&quot;</span><span class="s1">:</span>
154              <span class="s3">if </span><span class="s1">host_ip </span><span class="s3">is not None and </span><span class="s1">_ipaddress_match(value</span><span class="s3">, </span><span class="s1">host_ip):</span>
155                  <span class="s3">return</span>
156              <span class="s1">dnsnames.append(value)</span>
157      <span class="s3">if not </span><span class="s1">dnsnames:</span>
158          <span class="s2"># The subject is only checked when there is no dNSName entry</span>
159          <span class="s2"># in subjectAltName</span>
160          <span class="s3">for </span><span class="s1">sub </span><span class="s3">in </span><span class="s1">cert.get(</span><span class="s4">&quot;subject&quot;</span><span class="s3">, </span><span class="s1">()):</span>
161              <span class="s3">for </span><span class="s1">key</span><span class="s3">, </span><span class="s1">value </span><span class="s3">in </span><span class="s1">sub:</span>
162                  <span class="s2"># XXX according to RFC 2818, the most specific Common Name</span>
163                  <span class="s2"># must be used.</span>
164                  <span class="s3">if </span><span class="s1">key == </span><span class="s4">&quot;commonName&quot;</span><span class="s1">:</span>
165                      <span class="s3">if </span><span class="s1">_dnsname_match(value</span><span class="s3">, </span><span class="s1">hostname):</span>
166                          <span class="s3">return</span>
167                      <span class="s1">dnsnames.append(value)</span>
168      <span class="s3">if </span><span class="s1">len(dnsnames) &gt; </span><span class="s5">1</span><span class="s1">:</span>
169          <span class="s3">raise </span><span class="s1">CertificateError(</span>
170              <span class="s4">&quot;hostname %r &quot;</span>
171              <span class="s4">&quot;doesn't match either of %s&quot; </span><span class="s1">% (hostname</span><span class="s3">, </span><span class="s4">&quot;, &quot;</span><span class="s1">.join(map(repr</span><span class="s3">, </span><span class="s1">dnsnames)))</span>
172          <span class="s1">)</span>
173      <span class="s3">elif </span><span class="s1">len(dnsnames) == </span><span class="s5">1</span><span class="s1">:</span>
174          <span class="s3">raise </span><span class="s1">CertificateError(</span><span class="s4">&quot;hostname %r doesn't match %r&quot; </span><span class="s1">% (hostname</span><span class="s3">, </span><span class="s1">dnsnames[</span><span class="s5">0</span><span class="s1">]))</span>
175      <span class="s3">else</span><span class="s1">:</span>
176          <span class="s3">raise </span><span class="s1">CertificateError(</span>
177              <span class="s4">&quot;no appropriate commonName or subjectAltName fields were found&quot;</span>
178          <span class="s1">)</span>
