139      <span class="s2">def </span><span class="s1">cached_request(self</span><span class="s2">, </span><span class="s1">request):</span>

144          <span class="s1">cache_url = self.cache_url(request.url)</span>

174          <span class="s5">#</span>
175          <span class="s5"># Client can try to refresh the value by repeating the request</span>
176          <span class="s5"># with cache busting headers as usual (ie no-cache).</span>

186          <span class="s2">if not </span><span class="s1">headers </span><span class="s2">or </span><span class="s3">&quot;date&quot; </span><span class="s2">not in </span><span class="s1">headers:</span>

266      <span class="s2">def </span><span class="s1">cache_response(self</span><span class="s2">, </span><span class="s1">request</span><span class="s2">, </span><span class="s1">response</span><span class="s2">, </span><span class="s1">body=</span><span class="s2">None, </span><span class="s1">status_codes=</span><span class="s2">None</span><span class="s1">):</span>

268          Algorithm for caching requests.

271          &quot;&quot;&quot;</span>

273          <span class="s5">#                handle byte range requests</span>

277                  <span class="s3">&quot;Status code %s not in %s&quot;</span><span class="s2">, </span><span class="s1">response.status</span><span class="s2">, </span><span class="s1">cacheable_status_codes</span>
278              <span class="s1">)</span>
279              <span class="s2">return</span>

281          <span class="s1">response_headers = CaseInsensitiveDict(response.headers)</span>

283          <span class="s5"># If we've been given a body, our response has a Content-Length, that</span>
284          <span class="s5"># Content-Length is valid then we can check to see if the body we've</span>
285          <span class="s5"># been given matches the expected size, and if it doesn't we'll just</span>
286          <span class="s5"># skip trying to cache it.</span>

292          <span class="s1">):</span>
293              <span class="s2">return</span>
294  
295          <span class="s1">cc_req = self.parse_cache_control(request.headers)</span>
296          <span class="s1">cc = self.parse_cache_control(response_headers)</span>
297  
298          <span class="s1">cache_url = self.cache_url(request.url)</span>

300  
301          <span class="s5"># Delete it from the cache if we happen to have it stored there</span>
302          <span class="s1">no_store = </span><span class="s2">False</span>
303          <span class="s2">if </span><span class="s3">&quot;no-store&quot; </span><span class="s2">in </span><span class="s1">cc:</span>
304              <span class="s1">no_store = </span><span class="s2">True</span>

306          <span class="s2">if </span><span class="s3">&quot;no-store&quot; </span><span class="s2">in </span><span class="s1">cc_req:</span>
307              <span class="s1">no_store = </span><span class="s2">True</span>
308              <span class="s1">logger.debug(</span><span class="s3">'Request header has &quot;no-store&quot;'</span><span class="s1">)</span>
309          <span class="s2">if </span><span class="s1">no_store </span><span class="s2">and </span><span class="s1">self.cache.get(cache_url):</span>
310              <span class="s1">logger.debug(</span><span class="s3">'Purging existing cache entry to honor &quot;no-store&quot;'</span><span class="s1">)</span>
311              <span class="s1">self.cache.delete(cache_url)</span>
312          <span class="s2">if </span><span class="s1">no_store:</span>
313              <span class="s2">return</span>
314  
315          <span class="s5"># https://tools.ietf.org/html/rfc7234#section-4.1:</span>
316          <span class="s5"># A Vary header field-value of &quot;*&quot; always fails to match.</span>
317          <span class="s5"># Storing such a response leads to a deserialization warning</span>
318          <span class="s5"># during cache lookup and is not allowed to ever be served,</span>
319          <span class="s5"># so storing it can be avoided.</span>

322              <span class="s2">return</span>
323  
324          <span class="s5"># If we've been given an etag, then keep the response</span>
325          <span class="s2">if </span><span class="s1">self.cache_etags </span><span class="s2">and </span><span class="s3">&quot;etag&quot; </span><span class="s2">in </span><span class="s1">response_headers:</span>
326              <span class="s1">logger.debug(</span><span class="s3">&quot;Caching due to etag&quot;</span><span class="s1">)</span>
327              <span class="s1">self.cache.set(</span>
328                  <span class="s1">cache_url</span><span class="s2">, </span><span class="s1">self.serializer.dumps(request</span><span class="s2">, </span><span class="s1">response</span><span class="s2">, </span><span class="s1">body=body)</span>
329              <span class="s1">)</span>
330  
331          <span class="s5"># Add to the cache any 301s. We do this before looking that</span>
332          <span class="s5"># the Date headers.</span>
333          <span class="s2">elif </span><span class="s1">response.status == </span><span class="s4">301</span><span class="s1">:</span>

336  
337          <span class="s5"># Add to the cache if the response headers demand it. If there</span>
338          <span class="s5"># is no date header then we can't do anything about expiring</span>
339          <span class="s5"># the cache.</span>
340          <span class="s2">elif </span><span class="s3">&quot;date&quot; </span><span class="s2">in </span><span class="s1">response_headers:</span>
341              <span class="s5"># cache when there is a max-age &gt; 0</span>
342              <span class="s2">if </span><span class="s3">&quot;max-age&quot; </span><span class="s2">in </span><span class="s1">cc </span><span class="s2">and </span><span class="s1">cc[</span><span class="s3">&quot;max-age&quot;</span><span class="s1">] &gt; </span><span class="s4">0</span><span class="s1">:</span>
343                  <span class="s1">logger.debug(</span><span class="s3">&quot;Caching b/c date exists and max-age &gt; 0&quot;</span><span class="s1">)</span>
344                  <span class="s1">self.cache.set(</span>
345                      <span class="s1">cache_url</span><span class="s2">, </span><span class="s1">self.serializer.dumps(request</span><span class="s2">, </span><span class="s1">response</span><span class="s2">, </span><span class="s1">body=body)</span>
346                  <span class="s1">)</span>
347  
348              <span class="s5"># If the request can expire, it means we should cache it</span>
349              <span class="s5"># in the meantime.</span>
350              <span class="s2">elif </span><span class="s3">&quot;expires&quot; </span><span class="s2">in </span><span class="s1">response_headers:</span>
351                  <span class="s2">if </span><span class="s1">response_headers[</span><span class="s3">&quot;expires&quot;</span><span class="s1">]:</span>
352                      <span class="s1">logger.debug(</span><span class="s3">&quot;Caching b/c of expires header&quot;</span><span class="s1">)</span>
353                      <span class="s1">self.cache.set(</span>
354                          <span class="s1">cache_url</span><span class="s2">, </span><span class="s1">self.serializer.dumps(request</span><span class="s2">, </span><span class="s1">response</span><span class="s2">, </span><span class="s1">body=body)</span>
355                      <span class="s1">)</span>

357      <span class="s2">def </span><span class="s1">update_cached_response(self</span><span class="s2">, </span><span class="s1">request</span><span class="s2">, </span><span class="s1">response):</span>
358          <span class="s0">&quot;&quot;&quot;On a 304 we will get a new set of headers that we want to 
359          update our cached value with, assuming we have one. 
360   
361          This should only ever be called when we've sent an ETag and 
362          gotten a 304 as the response. 
363          &quot;&quot;&quot;</span>
364          <span class="s1">cache_url = self.cache_url(request.url)</span>
365  
366          <span class="s1">cached_response = self.serializer.loads(request</span><span class="s2">, </span><span class="s1">self.cache.get(cache_url))</span>
367  
368          <span class="s2">if not </span><span class="s1">cached_response:</span>
369              <span class="s5"># we didn't have a cached response</span>
370              <span class="s2">return </span><span class="s1">response</span>
371  
372          <span class="s5"># Lets update our headers with the headers from the new request:</span>
373          <span class="s5"># http://tools.ietf.org/html/draft-ietf-httpbis-p4-conditional-26#section-4.1</span>
374          <span class="s5">#</span>
375          <span class="s5"># The server isn't supposed to send headers that would make</span>
376          <span class="s5"># the cached body invalid. But... just in case, we'll be sure</span>
377          <span class="s5"># to strip out ones we know that might be problmatic due to</span>
378          <span class="s5"># typical assumptions.</span>
379          <span class="s1">excluded_headers = [</span><span class="s3">&quot;content-length&quot;</span><span class="s1">]</span>
380  
381          <span class="s1">cached_response.headers.update(</span>
382              <span class="s1">dict(</span>
383                  <span class="s1">(k</span><span class="s2">, </span><span class="s1">v)</span>
384                  <span class="s2">for </span><span class="s1">k</span><span class="s2">, </span><span class="s1">v </span><span class="s2">in </span><span class="s1">response.headers.items()</span>
385                  <span class="s2">if </span><span class="s1">k.lower() </span><span class="s2">not in </span><span class="s1">excluded_headers</span>
386              <span class="s1">)</span>
387          <span class="s1">)</span>
388  
389          <span class="s5"># we want a 200 b/c we have content via the cache</span>
390          <span class="s1">cached_response.status = </span><span class="s4">200</span>
391  
392          <span class="s5"># update our cache</span>
393          <span class="s1">self.cache.set(cache_url</span><span class="s2">, </span><span class="s1">self.serializer.dumps(request</span><span class="s2">, </span><span class="s1">cached_response))</span>
394  
395          <span class="s2">return </span><span class="s1">cached_response</span>
